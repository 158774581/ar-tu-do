128  // the lidar has a range of 270°, the algorithm use 180°

134  ROS_INFO_STREAM("right b: " << b);
135

142  // we use the car_length as a_c for now PLEASE CHANGE IN THE FUTURE

147  float c_d = a_b + a_c * std::sin(alpha);
148
